-- 김창섭 직원, 모든 정보
Select *
From Employee
Where EmpName = '김창섭'

-- 영업 부서의 층 수
Select Floor
From Department
Where DeptName = '영업';

-- 급여가 2500000 이상인 직원의 이름, 급여
Select EmpName, Salary
From Employee
Where Salary >= 2500000;

-- 과장 또는 대리의 이름, 급여, 직책
Select EmpName, Salary, Title
From Employee
Where Title = '과장' or Title = '대리';

-- 2번 부서가 아닌 직원 이름, 직책, 소속부서번호
Select EmpName, Title, Dno
From Employee
Where Dno <> 2;

-- 급여가 2000000 이상 2500000 이하인 직원 이름, 급여
Select EmpName, Salary
From Employee
Where Salary >= 2000000 and Salary <= 2500000;

-- 김씨 성 직원 이름, 소속부서번호
Select EmpName, Dno
From Employee
Where EmpName like '김%';

-- 1 또는 3번 부서에 속하는 직원 이름, 소속부서번호
Select EmpName, Dno
From Employee
Where Dno in (1, 3);

-- 모든 직원에 대하여 급여 오름차순으로 이름, 급여
Select EmpName, Salary
From Employee
Order by Salary Asc;
-- 오름차순이 디폴트여서 Asc 없어도 됌

-- 1번 부서에 근무하는 직원 급여 내림차순으로 이름, 급여
Select EmpName, Salary
From Employee
Where Dno = 1
Order by Salary Desc;
-- 내림차순은 Desc
