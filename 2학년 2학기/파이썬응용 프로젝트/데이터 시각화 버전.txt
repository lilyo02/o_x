3. 경찰청, 경찰청범죄통계_
import pandas as pd
import matplotlib.pyplot as plt

def customPct(pct) :
    return ('%.1f%%' % pct) if pct >= 15 else ''

data = pd.read_csv("교통범죄(2022년).csv", encoding = 'utf-8')

plt.rcParams.update({'font.family':'malgun gothic', 'font.size':7})

plt.title("교통범죄 중 음주운전의 비율")

plt.pie(data['발생건수(건)'], labels = data['상세죄종'], autopct = customPct)

plt.show()

->  2022년 경찰청에 접수된 형사 사건 중 교통범죄의 발생 건수에서 음주운전이 41.6%로 가장 많다.
